# Command with no args

FSW_CMD

==>
Sequence(
  Commands(
    Command(Stem,Args)
  )
)

# Command with two string args

FSW_CMD "hello" "world"
==>
Sequence(
  Commands(
    Command(Stem,Args(String,String))
  )
)

# Command with mixed args

FSW_CMD "hello" 10
==>
Sequence(Commands(
  Command(Stem,Args(String,Number))
))

# Command with mixed args and comment

FSW_CMD "hello" 10   # yay comment
==>
Sequence(Commands(
  Command(Stem,Args(String,Number),LineComment)
))

# 2 Commands

FSW_CMD "hello" "world"
FSW_CMD "hello" 10
==>
Sequence(Commands(
  Command(Stem,Args(String,String)),
  Command(Stem,Args(String,Number))
))

# Command with two sting args

FSW_CMD "TRUE" "FALSE" # end of line
# full line comment
==>
Sequence(Commands(
  Command(Stem,Args(String,String),LineComment),
  LineComment
))

# Command with enum args (disallowed in linter)

FSW_CMD TRUE FALSE
==>

Sequence(Commands(Command(Stem,Args(Enum,Enum))))

# Command with enum args (disallowed in linter) and comment

FSW_CMD TRUE FALSE
# Com
==>

Sequence(Commands(
  Command(Stem,Args(Enum,Enum)),
  LineComment
))

# Command with enum args (disallowed in linter) and more comments

FSW_CMD TRUE FALSE # Com 1
# Com
==>

Sequence(Commands(
  Command(Stem,Args(Enum,Enum),LineComment),
  LineComment
))
